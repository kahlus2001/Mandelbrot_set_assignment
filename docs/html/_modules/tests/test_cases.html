[docs]def test_random_numbers() -> None:
    """Tests if we add random numbers, the bounds are still respected.
    """
    for i in range(0, 100):
        x, y = random.uniform(-1, 1), random.uniform(-1, 1)
        assert -2.0 <= x + y <= 2.0
